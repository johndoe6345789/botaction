# jquery.min.js

## Overview

This file contains **jQuery library** - the classic DOM manipulation and AJAX library. Used for legacy functionality and some third-party integrations.

## File Information

- **Status**: Active library
- **Size**: ~88KB (minified)
- **Version**: 3.x (based on code patterns)
- **Type**: DOM manipulation library

## Core Features

### DOM Selection

```javascript
// Select elements
$('#myId')                    // By ID
$('.myClass')                 // By class
$('div')                      // By tag
$('div.myClass')              // Combined
$('[data-attr="value"]')      // By attribute
$('ul > li:first-child')      // CSS selectors

// Chaining
$('.items')
  .find('.active')
  .addClass('highlighted')
  .css('color', 'red');
```

### DOM Manipulation

```javascript
// Content
$('#el').html('<p>New content</p>');
$('#el').text('Plain text');
$('#el').val('input value');

// Attributes
$('#el').attr('href', 'url');
$('#el').data('key', 'value');
$('#el').removeAttr('disabled');

// Classes
$('#el').addClass('active');
$('#el').removeClass('active');
$('#el').toggleClass('active');
$('#el').hasClass('active');

// Styles
$('#el').css('color', 'red');
$('#el').css({ color: 'red', fontSize: '14px' });
$('#el').show();
$('#el').hide();
$('#el').toggle();
```

### Event Handling

```javascript

// On/off
$('#el').on('click', handler);

  // DOM loaded

```

### AJAX

```javascript
// GET request
$.get('/api/data', function(response) {
  console.log(response);
});

// POST request
$.post('/api/data', { name: 'value' }, function(response) {
  console.log(response);
});

// Full ajax
$.ajax({
  url: '/api/data',
  method: 'GET',
  data: { id: 123 },
  dataType: 'json',
  headers: { 'Authorization': 'Bearer token' },
  success: function(response) {
    console.log(response);
  },
  error: function(xhr, status, error) {
    console.error(error);
  },
  complete: function() {
    console.log('Request finished');
  }
});

// Promise-style
$.ajax('/api/data')
  .done(function(response) { })
  .fail(function(error) { })
  .always(function() { });
```

### Utilities

```javascript
// Arrays
$.each(array, function(index, item) { });
$.map(array, function(item, index) { return item * 2; });
$.grep(array, function(item) { return item > 5; });
$.inArray(value, array);
$.merge(array1, array2);

// Objects
$.extend(target, obj1, obj2);      // Shallow merge
$.extend(true, target, obj1);      // Deep merge

// Type checking
$.isArray(value);
$.isFunction(value);
$.isNumeric(value);
$.isPlainObject(value);
$.type(value);

// Strings
$.trim(string);
$.parseJSON(jsonString);
$.parseHTML(htmlString);
```

### Deferred/Promises

```javascript
// Create deferred
const deferred = $.Deferred();

// Resolve/reject
deferred.resolve(value);
deferred.reject(error);

// Promise methods
deferred.promise()
  .done(function(value) { })
  .fail(function(error) { })
  .always(function() { });

// When (wait for multiple)
$.when(promise1, promise2, promise3)
  .done(function(result1, result2, result3) { });
```

## Common Patterns in Sketchfab

### Legacy Popup Handling

```javascript
// Show popup
$('.popup').fadeIn(200);

// Close popup
$('.popup__close').on('click', function() {
  $(this).closest('.popup').fadeOut(200);
});

// Click outside to close
$(document).on('click', function(e) {
  if (!$(e.target).closest('.popup').length) {
    $('.popup').fadeOut(200);
  }
});
```

### Form Handling

```javascript
// Form submit
$('#myForm').on('submit', function(e) {
  e.preventDefault();
  
  const formData = $(this).serialize();
  
  $.ajax({
    url: $(this).attr('action'),
    method: 'POST',
    data: formData,
    success: function(response) {
      // Handle success
    }
  });
});

// Dynamic validation
$('input').on('blur', function() {
  const isValid = validateField($(this).val());
  $(this).toggleClass('error', !isValid);
});
```

### Lazy Loading

```javascript
// Intersection observer fallback
$('[data-src]').each(function() {
  const $img = $(this);
  
  if (isElementInViewport($img[0])) {
    $img.attr('src', $img.data('src'));
    $img.removeAttr('data-src');
  }
});
```
